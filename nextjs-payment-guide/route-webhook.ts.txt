import { NextResponse } from 'next/server';
import { PrismaClient } from '@prisma/client';

const globalForPrisma = global as unknown as { prisma: PrismaClient };
const prisma = globalForPrisma.prisma || new PrismaClient();
if (process.env.NODE_ENV !== 'production') globalForPrisma.prisma = prisma;

/**
 * App Router Handler: POST /api/webhook
 * Public endpoint that handles Instant Payment Notifications (IPN) sent by Paiement Pro.
 */
export async function POST(request: Request) {
  try {
    // 1. URL TOKEN SECURITY CHECK
    // Extract token from query params (e.g., /api/webhook?token=xxxx)
    const { searchParams } = new URL(request.url);
    const token = searchParams.get('token');
    
    const localWebhookSecret = process.env.PAYMENT_WEBHOOK_SECRET || 'SUPER_SECRET_TOKEN';
    
    if (!token || token !== localWebhookSecret) {
      console.warn("[WEBHOOK_SECURITY_ALERT]: Unauthorized webhook trigger attempt with invalid token.");
      return NextResponse.json({ error: "Unauthorized access token" }, { status: 401 });
    }

    // 2. Parse payload sent by Paiement Pro
    // Usually sent as standard form variables (urlencoded) or JSON payload depending on your merchant parameters.
    // We will support both parsing strategies gracefully.
    let payload: any = {};
    const contentType = request.headers.get("content-type") || "";

    if (contentType.includes("application/json")) {
      payload = await request.json();
    } else {
      // urlencoded parser
      const text = await request.text();
      const params = new URLSearchParams(text);
      payload = Object.fromEntries(params.entries());
    }

    console.log("[PAYMENT_PRO_WEBHOOK_PAYLOAD]:", payload);

    // Extract core parameters sent by Paiement Pro platform
    // NOTE: Typical parameters sent on webhook:
    // reference: unique order reference
    // status: payment status (e.g. 'SUCCESS' or 'FAILED')
    // amount: amount processed
    const reference = payload.reference || payload.referenceNumber || payload.ref;
    const status = payload.status || payload.payment_status;
    const amount = payload.amount;

    if (!reference) {
      return NextResponse.json({ error: "Reference de transaction manquante." }, { status: 400 });
    }

    // 3. Find corresponding order in Neon database
    const order = await prisma.order.findUnique({
      where: { referenceNumber: reference }
    });

    if (!order) {
      console.error(`[WEBHOOK_ERROR]: Aucune commande correspondante pour la reference: ${reference}`);
      return NextResponse.json({ error: "Order not found" }, { status: 404 });
    }

    // Avoid running double processing if webhook is re-sent or has already succeeded
    if (order.status === 'SUCCESS') {
      return NextResponse.json({ success: true, message: "Transaction déjà traitée." });
    }

    // 4. ULTIMATE SECURITY LAYER: Server-to-server transaction review (highly recommended)
    // To ensure a malicious attacker didn't just guess the URL token and forge status='SUCCESS', 
    // you should do a real-time verify call back to Paiement Pro's query service if they provide one,
    // OR verify critical figures like amount and merchant ID integrity.
    if (amount && Number(amount) !== order.amount) {
      console.error(`[WEBHOOK_FRAUD_DETECTED]: Le montant payé (${amount}) ne correspond pas au montant de la commande (${order.amount})`);
      await prisma.order.update({
        where: { id: order.id },
        data: { status: "FAILED" }
      });
      return NextResponse.json({ error: "Amount mismatch - Fraud flags triggered" }, { status: 400 });
    }

    // 5. Update transaction status in database
    if (status === 'SUCCESS' || status === 'APPROVED') {
      await prisma.order.update({
        where: { id: order.id },
        data: { status: "SUCCESS" }
      });
      
      console.log(`[PAYMENT_SUCCESS]: La commande #${order.id} a été payée avec succès via Paiement Pro !`);
      
      // Perform post-payment actions here:
      // - Enable premium access, credit tokens, trigger email confirmation, etc.
      // - Ex:
      // await prisma.user.update({
      //   where: { id: order.userId },
      //   data: { isPremium: true }
      // });

    } else {
      // Mark as failed
      await prisma.order.update({
        where: { id: order.id },
        data: { status: "FAILED" }
      });
      console.log(`[PAYMENT_FAILED]: Échec du paiement pour la commande #${order.id}.`);
    }

    // 6. Return a successful 200 OK block to acknowledge payload delivery to Paiement Pro
    return NextResponse.json({
      success: true,
      message: "Webhook traité avec succès"
    });

  } catch (error: any) {
    console.error("[WEBHOOK_RELIABILITY_CRASH]:", error);
    return NextResponse.json({
      success: false,
      error: error.message || "Internal server error occurred reading webhook"
    }, { status: 500 });
  }
}
