"use client";

import React, { useState, useEffect } from 'react';
// Import standard client-side Firebase Auth from your local setup
// import { auth } from '@/lib/firebase';
// import { onAuthStateChanged, User } from 'firebase/auth';

/**
 * CheckoutPage component: Simulates the client interface interacting with Next.js route handlers
 */
export default function CheckoutPage() {
  const [currentUser, setCurrentUser] = useState<any | null>(null);
  const [loadingUser, setLoadingUser] = useState(true);
  const [amount, setAmount] = useState<number>(1000); // 1000 CFA default
  const [operator, setOperator] = useState<string>('WAVECI'); // WAVECI, OMCI, MTNCI, MOOVCI, CARD
  const [phone, setPhone] = useState<string>('');
  const [isProcessing, setIsProcessing] = useState(false);
  const [errorMessage, setErrorMessage] = useState('');
  const [toastMessage, setToastMessage] = useState('');

  // 1. Listen to Firebase Client Authentication State
  useEffect(() => {
    // Replace with your real Firebase config path
    // const unsubscribe = onAuthStateChanged(auth, (user) => {
    //   setCurrentUser(user);
    //   setLoadingUser(false);
    // });
    // return () => unsubscribe();

    // Mock representation of Firebase listener for illustration:
    const mockUnsubscribe = () => {};
    setTimeout(() => {
      setCurrentUser({
        uid: 'firebase-user-uid-12345',
        email: 'jean.recruteur@gmail.com',
        displayName: 'Jean Kouassi'
      });
      setLoadingUser(false);
    }, 800);
    return () => mockUnsubscribe();
  }, []);

  // 2. Handle payment button click
  const handleCheckoutInit = async (e: React.FormEvent) => {
    e.preventDefault();
    setErrorMessage('');
    setIsProcessing(true);

    if (!currentUser) {
      setErrorMessage("Veuillez vous connecter avant d'initier un paiement.");
      setIsProcessing(false);
      return;
    }

    try {
      // Setup transaction context with user details
      const response = await fetch('/api/pay', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json'
        },
        body: JSON.stringify({
          amount: amount,
          userId: currentUser.uid, // PASSING the exact Firebase UID
          email: currentUser.email,
          firstName: currentUser.displayName?.split(' ')[0] || 'Client',
          lastName: currentUser.displayName?.split(' ').slice(1).join(' ') || 'Acheteur',
          phoneNumber: phone,
          channel: operator // Operator: WAVECI, OMCI, etc.
        })
      });

      const data = await response.json();

      if (!response.ok || !data.success) {
        throw new Error(data.error || "Erreur lors de l’initialisation de la transaction.");
      }

      setToastMessage("Redirection vers la passerelle sécurisée...");
      
      // 3. SECURE REDIRECT TO THE RETRIEVED PAIEMENT PRO GATEWAY URL
      if (data.paymentUrl) {
        window.location.href = data.paymentUrl;
      }

    } catch (err: any) {
      console.error(err);
      setErrorMessage(err.message || "Une erreur s'est produite lors de l'initialisation.");
      setIsProcessing(false);
    }
  };

  if (loadingUser) {
    return (
      <div className="flex h-screen items-center justify-center bg-gray-50 text-gray-700">
        <div className="text-center">
          <div className="h-8 w-8 animate-spin rounded-full border-4 border-indigo-600 border-t-transparent mx-auto"></div>
          <p className="mt-4 font-medium text-sm">Chargement de votre session...</p>
        </div>
      </div>
    );
  }

  return (
    <main className="min-h-screen bg-gray-50 py-12 px-4 sm:px-6 lg:px-8 font-sans">
      <div className="mx-auto max-w-md overflow-hidden rounded-2xl bg-white shadow-xl border border-gray-100">
        {/* Header decoration */}
        <div className="bg-indigo-600 px-6 py-8 text-center text-white">
          <h2 className="text-3xl font-extrabold tracking-tight">Espace Checkout</h2>
          <p className="mt-2 text-indigo-100 text-sm">Finalisez vos achats via Paiement Pro (Côte d'Ivoire)</p>
        </div>

        <div className="px-6 py-8">
          {/* Active Firebase Session indicators */}
          {currentUser ? (
            <div className="mb-6 rounded-lg bg-green-50 p-4 border border-green-200">
              <span className="block text-xs font-semibold text-green-800 uppercase tracking-wider">Session active (Firebase Auth)</span>
              <span className="block mt-1 text-sm font-semibold text-green-900">{currentUser.displayName || currentUser.email}</span>
              <span className="block text-xs text-green-700 mt-0.5">UID: {currentUser.uid}</span>
            </div>
          ) : (
            <div className="mb-6 rounded-lg bg-red-50 p-4 border border-red-200">
              <span className="block text-xs font-bold text-red-800 uppercase tracking-wider">Alerte connexion</span>
              <span className="block text-sm text-red-700 mt-1">Aucune session utilisateur trouvée.</span>
            </div>
          )}

          {errorMessage && (
            <div className="mb-6 rounded-lg bg-red-100 p-4 text-xs text-red-800 font-medium">
              ⚠️ {errorMessage}
            </div>
          )}

          {toastMessage && (
            <div className="mb-6 rounded-lg bg-indigo-50 p-4 text-xs text-indigo-800 font-medium animate-pulse">
              🚀 {toastMessage}
            </div>
          )}

          <form onSubmit={handleCheckoutInit} className="space-y-6">
            {/* Amount input */}
            <div>
              <label htmlFor="amount" className="block text-xs font-semibold text-gray-700 uppercase tracking-wide">Montant (FCFA)</label>
              <input
                id="amount"
                type="number"
                required
                value={amount}
                onChange={(e) => setAmount(Number(e.target.value))}
                min="100"
                className="mt-1 block w-full rounded-lg border border-gray-300 px-4 py-3 text-gray-900 shadow-sm focus:border-indigo-500 focus:outline-none focus:ring-1 focus:ring-indigo-500 sm:text-sm font-medium"
              />
            </div>

            {/* Operator Selection */}
            <div>
              <label htmlFor="operator" className="block text-xs font-semibold text-gray-700 uppercase tracking-wide">Moyen de paiement</label>
              <select
                id="operator"
                value={operator}
                onChange={(e) => setOperator(e.target.value)}
                className="mt-1 block w-full rounded-lg border border-gray-300 px-4 py-3 text-gray-900 shadow-sm focus:border-indigo-500 focus:outline-none focus:ring-1 focus:ring-indigo-500 sm:text-sm font-medium"
              >
                <option value="WAVECI">Wave (Côte d'Ivoire)</option>
                <option value="OMCI">Orange Money</option>
                <option value="MTNCI">MTN Mobile Money</option>
                <option value="MOOVCI">Moov Money</option>
                <option value="CARD">Carte Bancaire / Visa</option>
              </select>
            </div>

            {/* Phone entry (optional for Card checkout) */}
            {operator !== 'CARD' && (
              <div>
                <label htmlFor="phone" className="block text-xs font-semibold text-gray-700 uppercase tracking-wide">Numéro de téléphone mobile money</label>
                <input
                  id="phone"
                  type="tel"
                  required
                  placeholder="Ex: 0707070707"
                  value={phone}
                  onChange={(e) => setPhone(e.target.value)}
                  className="mt-1 block w-full rounded-lg border border-gray-300 px-4 py-3 text-gray-900 shadow-sm focus:border-indigo-500 focus:outline-none focus:ring-1 focus:ring-indigo-500 sm:text-sm font-medium"
                />
              </div>
            )}

            {/* Submit checkout initialization */}
            <button
              type="submit"
              disabled={isProcessing}
              className="flex w-full items-center justify-center rounded-lg bg-indigo-600 px-4 py-3.5 text-sm font-bold text-white shadow-md hover:bg-indigo-700 focus:outline-none focus:ring-2 focus:ring-indigo-500 focus:ring-offset-2 disabled:bg-gray-400 disabled:cursor-not-allowed transition duration-150 ease-in-out"
            >
              {isProcessing ? (
                <>
                  <div className="mr-2 h-4 w-4 animate-spin rounded-full border-2 border-white border-t-transparent"></div>
                  Génération du lien...
                </>
              ) : `Payer ${amount.toLocaleString()} FCFA`}
            </button>
          </form>
        </div>
      </div>
    </main>
  );
}
